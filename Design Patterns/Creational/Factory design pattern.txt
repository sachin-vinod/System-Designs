#include <iostream>
using namespace std;

class Vehical{
    public:
        virtual void getDetail() = 0;
};

class TwoWheeler : public Vehical{
    public:
    void getDetail() override{
        cout<<"I am two wheeler";
    }
};

class ThreeWheeler : public Vehical{
    public:
        void getDetail() override{
            cout<<"I am Three Wheeler";
        };
};

class Factory{
    public:
        static Vehical* getTypeOfVehical(string type){
            if(type=="Two Wheeler"){
                return new TwoWheeler();
            }
            else if(type=="Three Wheeler"){
                return new ThreeWheeler();
            }
            else{
                return NULL;
            }
        }
};


int main() {
    Vehical *vehical=Factory::getTypeOfVehical("Three Wheeler");
    vehical->getDetail();
    return 0;
}
